---
title: 'useNFC'
description: 'Scan and read NFC tags'
---

React hook that provides NFC tag scanning capabilities with support for NDEF messages and tag discovery events.

## Import

```typescript
import { useNFC } from 'node-fullykiosk';
```

## Usage

```tsx
import { useNFC } from 'node-fullykiosk';
import { useState } from 'react';

function NFCScanner() {
  const [tagData, setTagData] = useState(null);
  
  const { isScanning, start, stop } = useNFC({
    onNdef: (serial, message, data) => {
      console.log('NDEF discovered:', { serial, message, data });
      setTagData({ serial, message, data });
    },
    onNfcTagDiscovered: (serial, type, message, data) => {
      console.log('Tag discovered:', { serial, type, message, data });
    },
    onNfcTagRemoved: (serial) => {
      console.log('Tag removed:', serial);
      setTagData(null);
    }
  });

  return (
    <div>
      <h2>NFC Scanner</h2>
      <p>Status: {isScanning ? 'Scanning...' : 'Stopped'}</p>
      <button onClick={start}>Start Scanning</button>
      <button onClick={stop}>Stop Scanning</button>
      {tagData && (
        <div>
          <h3>Tag Data:</h3>
          <p>Serial: {tagData.serial}</p>
          <p>Message: {tagData.message}</p>
        </div>
      )}
    </div>
  );
}
```

## Parameters

<ParamField path="config" type="object" optional>
  Configuration object for NFC event handlers.

  <Expandable title="config properties">
    <ResponseField name="onNdef" type="(serial: string, message: string, data: string) => void" optional>
      Callback invoked when an NDEF message is discovered on an NFC tag.
      - `serial`: The tag's serial number
      - `message`: The NDEF message content
      - `data`: Additional data from the tag
    </ResponseField>

    <ResponseField name="onNfcTagDiscovered" type="(serial: string, type: string, message: string, data: string) => void" optional>
      Callback invoked when any NFC tag is discovered.
      - `serial`: The tag's serial number
      - `type`: The type of NFC tag
      - `message`: Message content from the tag
      - `data`: Additional data from the tag
    </ResponseField>

    <ResponseField name="onNfcTagRemoved" type="(serial: string) => void" optional>
      Callback invoked when an NFC tag is removed from the reader.
      - `serial`: The serial number of the removed tag
    </ResponseField>
  </Expandable>
</ParamField>

## Return Value

<ResponseField name="isScanning" type="boolean">
  Indicates whether NFC scanning is currently active.
</ResponseField>

<ResponseField name="start" type="() => void">
  Function to start NFC scanning. Throws an error if Fully Kiosk is not available.
</ResponseField>

<ResponseField name="stop" type="() => void">
  Function to stop NFC scanning. Throws an error if Fully Kiosk is not available.
</ResponseField>

## Behavior

- Automatically registers NFC event handlers on mount
- Requires Fully Kiosk to be available (shows alert if not)
- Events are triggered when tags are detected, discovered, or removed
- Scanning must be manually started with `start()` and stopped with `stop()`
- Event listeners persist for the lifetime of the component

## Example: NFC Tag Reader

```tsx
import { useNFC } from 'node-fullykiosk';
import { useState } from 'react';

function NFCReader() {
  const [tags, setTags] = useState([]);
  const [currentTag, setCurrentTag] = useState(null);

  const { isScanning, start, stop } = useNFC({
    onNfcTagDiscovered: (serial, type, message, data) => {
      const tag = {
        serial,
        type,
        message,
        data,
        timestamp: new Date().toISOString()
      };
      setCurrentTag(tag);
      setTags(prev => [tag, ...prev].slice(0, 10)); // Keep last 10
    },
    onNfcTagRemoved: (serial) => {
      if (currentTag?.serial === serial) {
        setCurrentTag(null);
      }
    }
  });

  return (
    <div style={{ padding: '20px' }}>
      <div style={{
        display: 'flex',
        justifyContent: 'space-between',
        alignItems: 'center',
        marginBottom: '20px'
      }}>
        <h2>NFC Tag Reader</h2>
        <button
          onClick={isScanning ? stop : start}
          style={{
            padding: '10px 20px',
            backgroundColor: isScanning ? '#f44336' : '#4CAF50',
            color: 'white',
            border: 'none',
            borderRadius: '6px',
            cursor: 'pointer'
          }}
        >
          {isScanning ? 'Stop Scanning' : 'Start Scanning'}
        </button>
      </div>

      {currentTag && (
        <div style={{
          padding: '16px',
          backgroundColor: '#e3f2fd',
          borderRadius: '8px',
          marginBottom: '20px'
        }}>
          <h3 style={{ marginTop: 0 }}>Current Tag</h3>
          <p><strong>Serial:</strong> {currentTag.serial}</p>
          <p><strong>Type:</strong> {currentTag.type}</p>
          <p><strong>Message:</strong> {currentTag.message}</p>
          <p><strong>Data:</strong> {currentTag.data}</p>
        </div>
      )}

      <div>
        <h3>Scan History</h3>
        {tags.length === 0 ? (
          <p style={{ color: '#666' }}>No tags scanned yet</p>
        ) : (
          <ul style={{ listStyle: 'none', padding: 0 }}>
            {tags.map((tag, index) => (
              <li key={index} style={{
                padding: '12px',
                backgroundColor: '#f5f5f5',
                borderRadius: '6px',
                marginBottom: '8px'
              }}>
                <strong>{tag.serial}</strong> - {tag.type}
                <br />
                <small style={{ color: '#666' }}>
                  {new Date(tag.timestamp).toLocaleString()}
                </small>
              </li>
            ))}
          </ul>
        )}
      </div>
    </div>
  );
}
```

## Example: NDEF Message Handler

```tsx
import { useNFC } from 'node-fullykiosk';
import { useState } from 'react';

function NDEFMessageHandler() {
  const [ndefData, setNdefData] = useState(null);

  const { isScanning, start, stop } = useNFC({
    onNdef: (serial, message, data) => {
      // Parse NDEF message
      try {
        const parsedData = JSON.parse(data);
        setNdefData({
          serial,
          message,
          parsedData,
          raw: data
        });
      } catch (e) {
        setNdefData({
          serial,
          message,
          parsedData: null,
          raw: data
        });
      }
    }
  });

  return (
    <div>
      <h2>NDEF Message Handler</h2>
      
      <div style={{ marginBottom: '20px' }}>
        {!isScanning ? (
          <button onClick={start}>Start Scanning</button>
        ) : (
          <button onClick={stop}>Stop Scanning</button>
        )}
      </div>

      {ndefData && (
        <div style={{
          padding: '16px',
          backgroundColor: '#fff3e0',
          borderRadius: '8px'
        }}>
          <h3>NDEF Message</h3>
          <p><strong>Serial:</strong> {ndefData.serial}</p>
          <p><strong>Message:</strong> {ndefData.message}</p>
          {ndefData.parsedData && (
            <pre style={{
              backgroundColor: '#f5f5f5',
              padding: '12px',
              borderRadius: '4px',
              overflow: 'auto'
            }}>
              {JSON.stringify(ndefData.parsedData, null, 2)}
            </pre>
          )}
        </div>
      )}
    </div>
  );
}
```

## Example: Access Control System

```tsx
import { useNFC } from 'node-fullykiosk';
import { useState, useEffect } from 'react';

function AccessControl() {
  const [accessGranted, setAccessGranted] = useState(false);
  const [message, setMessage] = useState('');
  const authorizedTags = ['TAG001', 'TAG002', 'TAG003'];

  const { isScanning, start, stop } = useNFC({
    onNfcTagDiscovered: (serial, type, message, data) => {
      if (authorizedTags.includes(serial)) {
        setAccessGranted(true);
        setMessage(`Welcome! Access granted for tag ${serial}`);
      } else {
        setAccessGranted(false);
        setMessage(`Access denied for tag ${serial}`);
      }
    },
    onNfcTagRemoved: (serial) => {
      setAccessGranted(false);
      setMessage('Tag removed. Please scan your tag.');
    }
  });

  useEffect(() => {
    // Auto-start scanning on mount
    start();
    return () => stop();
  }, []);

  return (
    <div style={{
      display: 'flex',
      flexDirection: 'column',
      alignItems: 'center',
      justifyContent: 'center',
      minHeight: '100vh',
      backgroundColor: accessGranted ? '#4CAF50' : '#f44336',
      color: 'white',
      transition: 'background-color 0.3s'
    }}>
      <div style={{ fontSize: '64px', marginBottom: '20px' }}>
        {accessGranted ? '✓' : '🔒'}
      </div>
      <h1>{accessGranted ? 'Access Granted' : 'Access Denied'}</h1>
      <p style={{ fontSize: '18px', marginTop: '16px' }}>
        {message || 'Please scan your NFC tag'}
      </p>
      {isScanning && (
        <p style={{ marginTop: '20px', opacity: 0.8 }}>
          Scanning for tags...
        </p>
      )}
    </div>
  );
}
```

## Notes

- Requires NFC hardware support on the device
- Requires appropriate Android permissions for NFC access
- The hook will show an alert if Fully Kiosk is not available
- Event handlers are registered once on mount and persist throughout the component lifecycle
- Remember to call `start()` to begin scanning - it doesn't start automatically
